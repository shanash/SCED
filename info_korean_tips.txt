  -- load card data
  ---@param cardMetadata table  Contains card metadata
  ---@return string|nil
  ---@return string|nil
  WebInterface.loadCardData = function(cardMetadata)
    function handleRequest(cardId)
      local adbInfo, isDone = nil, false
      local conf = configuration or getConfiguration()

      WebRequestLib.get({
        conf.ArkhamDB_api_uri,
        conf.cards,
        cardId
      }, function(result)
        local raw = result or ""
        local fixed = (fixUtf16String and fixUtf16String(raw)) or raw
        local ok, decoded = pcall(JSON.decode, fixed)
        if ok and type(decoded) == "table" then
          adbInfo = decoded
        end
        isDone = true
      end,
      function()
        isDone = true
      end)
      
      repeat coroutine.yield()
      until isDone
      
      return adbInfo
    end

    local cardId = cardMetadata.id
    local isBeta = string.find(string.sub(cardId, 1, 1), "B")
    
    if (isBeta) then
      return nil, nil
    end
    
    local cardName = nil
    local cardText = nil
    local baseId = nil
    
    if (string.find(cardId, "%-")) then
      baseId = string.sub(cardId, 1, string.find(cardId, "%-") - 1)
    else
      baseId = cardId
    end
    
    local adbInfo = nil
    local isMini = string.find(cardId, "%-m")
    local isTaboo = string.find(cardId, "%-t")
    local isCustom = string.find(cardId, "%-c")
    local hasPf = string.find(cardId, "%-p") and (not string.find(cardId, "%-pb"))
    local hasPb = string.find(cardId, "%-p") and (not string.find(cardId, "%-pf"))
    local isConnected = string.find(cardId, "%-b")
    local connectedId = nil
    
    if (isConnected) then
      connectedId = baseId
      if (string.find(cardId, "%-b1")) then
        baseId = (tonumber(connectedId) + 1) .. "a"
      elseif (string.find(cardId, "%-b2")) then
        baseId = (tonumber(connectedId) + 1) .. "b"
      end
    end
    
    adbInfo = handleRequest(baseId)
    
    if (adbInfo and adbInfo.name ~= nil) then
      if (not isMini and adbInfo.is_unique) then
        cardName = "* " .. adbInfo.name
      else
        cardName = adbInfo.name
      end
      
      if (not isMini) then
        if (hasPf) then
          cardName = cardName .. ": 평행"
          if (hasPb) then
            cardName = cardName .. "/평행"
          else
            cardName = cardName .. "/일반"
          end
        elseif (hasPb) then
            cardName = cardName .. ": 일반/평행"
        end
        
        if (adbInfo.xp ~= nil and adbInfo.xp > 0) then
          cardName = cardName .. " (" .. adbInfo.xp .. ")"
        end
        
        if (isTaboo) then
          cardName = cardName .. " (금기)"
        end
        
        if (adbInfo ~= nil and adbInfo.text ~= nil) then
          local tidx = 0
          local textLine = {}
          local sideInfo = nil
          local adbInfoParallel = nil
          local requestParallel = nil
          local adbInfoLinked = nil
          local requestLinked = nil
          
          if (adbInfo.alternated_by ~= nil) then
            local parallelCode = nil
            for k, v in pairs(adbInfo.alternated_by) do
              altCode = (tonumber(v) or 0)
              if (altCode > 90000 and altCode <= 98000) then
                parallelCode = v
                break
              end
            end
            
            if (parallelCode ~= nil) then
              adbInfoParallel = handleRequest(parallelCode)
            end
          end
          
          if (adbInfo.linked_to_code ~= nil) then
            adbInfoLinked = handleRequest(adbInfo.linked_to_code)
          end
          
          if (adbInfo.subname ~= nil) then
            if (adbInfoLinked ~= nil and adbInfoLinked.subname ~= nil and adbInfo.subname ~= adbInfoLinked.subname) then
              textLine[tidx] = adbInfo.subname .. " (A면)\n-------"
            else
              textLine[tidx] = adbInfo.subname .. "\n-------"
            end
            tidx = tidx +1
          end
          
          if (isCustom and adbInfo.customization_text ~= nil) then
            textLine[tidx] = adbInfo.customization_text
            tidx = tidx +1
          else
            local textBack = nil
            local textFront = nil
            local traitsFront = nil
            
            if (hasPf and adbInfoParallel ~= nil) then
              textFront = adbInfoParallel.text
              traitsFront = adbInfoParallel.traits
            elseif (adbInfo.text ~= nil) then
              textFront = adbInfo.text
              traitsFront = adbInfo.traits
            end
            
            if (hasPb and adbInfoParallel ~= nil) then
              textBack = adbInfoParallel.back_text
            elseif (adbInfo.back_text ~= nil) then
              textBack = adbInfo.back_text
            end
            
            if (traitsFront ~= nil) then
              textLine[tidx] = traitsFront .. "\n\n" .. textFront
            else
              textLine[tidx] = textFront
            end
            tidx = tidx +1
            
            if (textBack ~= nil) then
              textLine[tidx] = "\n==========\n\n" .. textBack
              tidx = tidx +1
            end
            
            if (adbInfoLinked ~= nil) then
              if (adbInfoLinked ~= nil and adbInfoLinked.subname ~= nil and adbInfo.subname ~= adbInfoLinked.subname) then
                textLine[tidx] = "\n==========\n\n" .. adbInfoLinked.subname .. " (B면)\n-------"
                tidx = tidx +1
              end
              
              if (adbInfoLinked.traits ~= nil) then
                textLine[tidx] = adbInfoLinked.traits .. "\n\n" .. adbInfoLinked.text
              else
                textLine[tidx] = adbInfoLinked.text
              end
              tidx = tidx +1
            end
          end
          
          if (isTaboo) then
            textLine[tidx] = "\n==========\n" .. "금기 내용은 제공되지 않습니다. 원하는 버전의 금기 내용은 아컴허브에서 확인하십시오. (카드 이미지는 항상 최신의 영문 버전만 제공됩니다.)"
            tidx = tidx +1
          end
          
          for idx=0, tidx do
            if (textLine[idx] == nil) then
              break
            end
            
            if (cardText ~= nil) then
              cardText = cardText .. "\n" .. textLine[ idx]
            else
              cardText = textLine[idx]
            end
          end
          
          if (cardText) then
            cardText = string.gsub(cardText, "<b>", "[b]")
            cardText = string.gsub(cardText, "</b>", "[/b]")
            cardText = string.gsub(cardText, "<i>", "[i]")
            cardText = string.gsub(cardText, "</i>", "[/i]")
            cardText = string.gsub(cardText, "%[%[", "[i]")
            cardText = string.gsub(cardText, "%]%]", "[/i]")
            cardText = string.gsub(cardText, "%[reaction%]", "[반응격발]")
            cardText = string.gsub(cardText, "%[action%]", "[행동격발]")
            cardText = string.gsub(cardText, "%[free%]", "[자유격발]")
            cardText = string.gsub(cardText, "%[elder_sign%]", "[고대표식]")
            cardText = string.gsub(cardText, "%[skull%]", "[해골]")
            cardText = string.gsub(cardText, "%[cultist%]", "[추종자]")
            cardText = string.gsub(cardText, "%[tablet%]", "[석판]")
            cardText = string.gsub(cardText, "%[elder_thing%]", "[옛것]")
            cardText = string.gsub(cardText, "%[auto_fail%]", "[자동실패]")
            cardText = string.gsub(cardText, "%[curse%]", "[저주]")
            cardText = string.gsub(cardText, "%[bless%]", "[축복]")
            cardText = string.gsub(cardText, "%[willpower%]", "[의지]")
            cardText = string.gsub(cardText, "%[intellect%]", "[지식]")
            cardText = string.gsub(cardText, "%[combat%]", "[힘]")
            cardText = string.gsub(cardText, "%[agility%]", "[민첩]")
            cardText = string.gsub(cardText, "%[wild%]", "[만능]")
            cardText = string.gsub(cardText, "%[mystic%]", "[신비주의자]")
            cardText = string.gsub(cardText, "%[rogue%]", "[무법자]")
            cardText = string.gsub(cardText, "%[seeker%]", "[탐구자]")
            cardText = string.gsub(cardText, "%[survivor%]", "[생존자]")
            cardText = string.gsub(cardText, "%[guardian%]", "[수호자]")
            cardText = string.gsub(cardText, "%[neutral%]", "[중립]")
            cardText = string.gsub(cardText, "%[health%]", "[체력]")
            cardText = string.gsub(cardText, "%[sanity%]", "[정신력]")
            cardText = string.gsub(cardText, "%[per_investigator%]", "[x조사자수]")
          end
        end
      end
    end
    
    return cardName, cardText
  end

4. 잘 작동하는지 테스트 해보려면 아래처럼 해보면 됨.

Build All 써 있던 그 오브젝트 내에서 적당한 칸에 숫자를 입력하고 적당한 버튼을 누르면 카드들이 로드되는데

각 카드에 마우스를 올려두고 3~5초 정도? 기다렸을 때, 툴팁 부분에 원하는 결과가 떠야 함.



근데 이 3~5초되는 이 툴팁이 뜨는 시간조차 줄이려면 아래 중 하나로 적용해보면 됨 (둘 중 하나만!!!)

① 메뉴 - 구성 - 인터페이스 - 기타/설정 - 구성 요소 툴팁 지연 항목을 0으로 변경

② ` 버튼 누르고 콘솔창에 ‘component_tooltip_delay 0’ 입력 (`: 물결표시 아래 따옴표 or 그레이브)



해당 웹에 있는 데이터를 그대로 가져오는 거라, 일부는 원하는 결과가 안 뜨기도 함.

아, private 인지, public 인지 잘 확인하고 테스트 바람.
